605 static inline bool
606 sol_json_loop_iterate_generic(struct sol_json_scanner *scanner, struct sol_json_token *token,
607  enum sol_json_type end_type, enum sol_json_loop_status *reason)
608 {
609  if (*reason != SOL_JSON_LOOP_REASON_OK)
610  return false;
611 
612  if (!sol_json_scanner_next(scanner, token)) {
613  *reason = SOL_JSON_LOOP_REASON_INVALID;
614  return false;
615  }
616 
617  if (sol_json_token_get_type(token) == end_type) {
618  *reason = SOL_JSON_LOOP_REASON_OK;
619  return false;
620  }
621 
622  if (sol_json_token_get_type(token) == SOL_JSON_TYPE_ELEMENT_SEP) {
623  if (!sol_json_scanner_next(scanner, token)) {
624  *reason = SOL_JSON_LOOP_REASON_INVALID;
625  return false;
626  }
627  }
628 
629  return true;
630 }
