831 static inline int
832 sol_buffer_trim(struct sol_buffer *buf)
833 {
834  size_t new_size;
835 
836  if (!buf)
837  return -EINVAL;
838 
839  if (buf->flags & SOL_BUFFER_FLAGS_NO_NUL_BYTE)
840  new_size = buf->used;
841  else if (buf->used == SIZE_MAX)
842  return -EOVERFLOW;
843  else
844  new_size = buf->used + 1;
845 
846  if (new_size == buf->capacity)
847  return 0;
848 
849  return sol_buffer_resize(buf, new_size);
850 }
