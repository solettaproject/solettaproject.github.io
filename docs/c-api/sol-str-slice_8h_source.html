344 static inline struct sol_str_slice

346 {
347  struct sol_str_slice copy = slice;
348 
349  while (copy.len != 0) {
350  uint8_t c = copy.data[copy.len - 1];
351  if (isspace(c))
352  copy.len--;
353  else
354  break;
355  }
356 
357  if (slice.len - copy.len)
358  return copy;
359 
360  return slice;
