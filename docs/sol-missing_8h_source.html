70 accept4(int sockfd, struct sockaddr *addr, socklen_t *len, int flags)
71 {
72  int fl, fd = accept(sockfd, addr, len);
73 
74  if (fd < 0)
75  return fd;
76 
77  fl = fcntl(fd, F_GETFD);
78  if (fl == -1)
79  goto err;
80 
81  fl |= flags;
82  if (fcntl(fd, F_SETFD, fl) == -1)
83  goto err;
84 
85  return fd;
86 
87 err:
88  close(fd);
89  return -1;
